export const DRAG_COEFFICIENT = 0.1   // Drag factor
export const LIFT_COEFFICIENT = 0.5   // Lift factor

export class FlightDynamics {
  public applyThrust(forward, backward, left, right): void {
    // Clamp inputs to [-1, 1]
    // Apply thrust in local coordinate system
    this.body.applyLocalForce(thrustLocal, new CANNON.Vec3(0, 0, 0))
  }

  public applyDrag(): void {
    // Drag = -dragCoeff * velocity * |velocity|
    const dragMagnitude = this.dragCoefficient * speed
    const dragForce = new CANNON.Vec3(
      -velocity.x * dragMagnitude,
      -velocity.y * dragMagnitude,
      -velocity.z * dragMagnitude
    )
    this.body.applyForce(dragForce, this.body.position)
  }

  public applyLift(): void {
    // Lift = liftCoeff * forwardSpeed (upward)
    const liftForce = new CANNON.Vec3(0, this.liftCoefficient * forwardSpeed, 0)
    this.body.applyForce(liftForce, this.body.position)
  }

  public update(forward, backward, left, right): void {
    this.resetForces()
    this.applyThrust(forward, backward, left, right)
    this.applyDrag()
    this.applyLift()
  }
}

export const FIXED_TIME_STEP = 1 / 60 // 60 FPS
export const MAX_SUB_STEPS = 3 // Catch-up limit

export class PhysicsUpdateLoop {
  public calculateDeltaTime(currentTime: number, lastTime: number): number {
    return (currentTime - lastTime) / 1000
  }

  public updatePhysics(deltaTime: number): void {
    let dt = deltaTime

    // Handle invalid values
    if (isNaN(dt) || !isFinite(dt)) dt = 0
    if (dt < 0) dt = 0

    this.world.step(FIXED_TIME_STEP, dt, MAX_SUB_STEPS)
  }

  public syncMeshToBody(): void {
    this.mesh.position.copy(this.body.position)
    this.mesh.quaternion.copy(this.body.quaternion)
  }

  public update(deltaTime: number): void {
    this.updatePhysics(deltaTime)
    this.syncMeshToBody()
    this.frameCount++
    this.elapsedTime += deltaTime
  }

  // start(), stop(), reset(), isRunning(), getElapsedTime(), getFrameCount()
}

Task (⬇️) → Feature (⬆️) → Block (⬆️) → Product (⬆️)
70 Tasks → 14 Features → 4 Blocks → 1 Product

### 핵심 교훈

**"개발 중 발견한 필요성은 즉시 반영한다"**
- 초기 계획(3 Blocks)이 부족함을 발견
- 고정된 구조에 얽매이지 않고 유연하게 확장
- Clean Architecture 원칙 준수 (레이어 분리)

**"PRD와 설계 문서는 살아있는 문서"**
- 개발 중 불일치 발견 시 즉시 동기화
- 문서화가 개발 품질을 보장
